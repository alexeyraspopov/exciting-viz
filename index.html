amplify cognition.

It's all about describing abstract values (weight, size, type, category, age, etc)
using visual values (coords, size, shape,
color, pattern, etc).

Scale — a plain function that takes an abstract value of data, and returns a visual value. 

        // domain is abstract, range is visual
        function scaleLinear(domain, range) {
          return x => interpolate(range, normalize(domain, x));
        }
      
        // [N, N] -> [0, 1]
        function normalize(range, x) {
          let k = range[1] - range[0];
          let b = range[0];
          return (x - b) / k;
        }
        // [0, 1] -> [N, N]
        function interpolate(range, t) {
          let k = range[1] - range[0];
          let b = range[0];
          return b + k * t;
        }
      
        // scale represented as a single formula
        function scale([d0, d1], [r0, r1], x) {
          return r0 + (r1 - r0) * (x - d0) / (d1 - d0);
        }

            let prefix = i === 0 ? 'M' : 'L';
            let point = prefix + x(d, i) + ',' + y(d, i);
